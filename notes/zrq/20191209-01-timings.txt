# Create a container to work with.
#[user@desktop]

    docker run \
        --rm \
        --tty \
        --interactive \
        --hostname openstacker \
        --volume "${HOME}/clouds.yaml:/etc/openstack/clouds.yaml" \
        atolmis/openstack-client \
        bash



# -----------------------------------------------------
# Set the project and cluster names.
#[user@openstacker]

    cloudname=gaia-prod
    clustername=Augustus

# -----------------------------------------------------
# List our clusters.
#[user@openstacker]

    time \
        openstack \
            --os-cloud "${cloudname:?}" \
            coe cluster list

    >   real    0m4.202s
    >   user    0m1.114s
    >   sys     0m0.129s


# -----------------------------------------------------
# Get our keypair name.
#[user@openstacker]

    time \
        keyname=$(
            openstack \
            --os-cloud "${cloudname:?}" \
                keypair list \
                    --format json \
            | jq -r '.[0] | .Name'
            )

    >   real    0m2.549s
    >   user    0m1.186s
    >   sys     0m0.120s

    echo "Key name [${keyname}]"

    >   Key name [zrq-gaia-keypair]

# -----------------------------------------------------
# Get the small flavor ID.
#[user@openstacker]

    time \
        flavorid=$(
            openstack \
                --os-cloud "${cloudname:?}" \
                flavor list \
                    --format json \
            | jq -r '.[] | select(.Name == "general.v1.small") | .ID'
            )

    >   real    0m2.810s
    >   user    0m1.262s
    >   sys     0m0.122s

    echo "Flavor ID [${flavorid}]"

    >   Flavor ID [20061eba-9e88-494c-95a3-41ed77721244]


# -----------------------------------------------------
# Get the uuid for the Octavia template.
#[user@openstacker]

    time \
        templateuuid=$(
            openstack \
                --os-cloud "${cloudname:?}" \
                coe cluster template list \
                    --format json \
            | jq -r '.[] | select(.name | test("octavia")) | .uuid'
            )

    >   real    0m4.084s
    >   user    0m1.210s
    >   sys     0m0.151s

    echo "Template [${templateuuid}]"

    >   Template [41437926-8461-4fdf-ac1b-ff97325a79f8]


# -----------------------------------------------------
# Create a new cluster, using unrestricted credentials.
#[user@openstacker]

    time \
        openstack \
            --os-cloud "${cloudname:?}-super" \
            coe cluster create \
                --keypair "${keyname:?}" \
                --flavor  "${flavorid:?}" \
                --node-count 4 \
                --master-count 2 \
                --master-flavor "${flavorid:?}" \
                --cluster-template "${templateuuid:?}" \
                "${clustername:?}"

    >   Request to create cluster 7776718b-02f1-43dd-b029-6f10b6592de0 accepted

    >   real    0m6.186s
    >   user    0m0.939s
    >   sys     0m0.102s

#[user@openstacker]

        openstack \
            --os-cloud "${cloudname:?}" \
